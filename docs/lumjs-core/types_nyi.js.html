/**
 * A placeholder function for when something is not implemented.
 * 
 * @param {boolean} [fatal=true] - If `true` throw an `Error`.
 *   If `false` just uses `console.error()` instead.
 * @returns {void}
 */
 function NYI(fatal=true) 
 { 
   const msg = "Not yet implemented";
   if (fatal)
     throw new Error(msg);
   else
     console.error(msg);
 }

 module.exports = NYI;
